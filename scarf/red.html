Sporky has spent quite a lot of money on our adventure today. I am getting a little bit hungry and not too sure if I should buy some food.

The code below calculates Sporky's total spendings so far today. To decide whether sporky should buy something to eat, decipher the output and select what he will buy. 
     import java.util.Scanner;

     public class MyClass {
          public static void main(String[] args) {
               Scanner inputCost = null;
               double[] cost = new double[2];
               String travelCost;
               double totalCost = 0;
               int i;
          
               travelCost = "80.5 35.00";
          
               inputCost = new Scanner(travelCost);
          
               for(i=0; i < 2; ++i) {
                    cost[i] = inputCost.nextDouble();
                    totalCost = totalCost + cost[i];
               }
          
               System.out.printf("Sporky has spent $%.2f", totalCost);
               System.out.println();
               System.out.println();
          
               //FOOD SPORKY BUYS
               System.out.print("With Sporky's remaining money, ");
               if (totalCost <= 115.50) {
                      System.out.println("Sporky will get a steak meal.");
               }
               else if (totalCost <= 121.00) {
                    System.out.println("Sporky will get a pepperoni pizza.");
               }
               else if (totalCost <= 125.50) {
                    System.out.println("Sporky will get a cheeseburger.");
               }
               else {
                    System.out.println("he is too broke to buy any food.");
               } 
          }
     }
